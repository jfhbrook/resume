\begin{project}
  \projectname{The Determination of Anisotropic Thermal Conductivity with Thermal
    Needle Probe Measurements}
  \dates[May 2010]{May 2011}

  \begin{projectdescription}
    A master's thesis on the theory and viability of measuring anisotropic
    thermal conductivity in snow with needle probes, including 3D finite element
    models in COMSOL and MATLAB; an analytical model; a prototype experimental
    procedure for benchmarking the technique; and the first attempt at using the
    technique against real snow
  \end{projectdescription}
\end{project}
